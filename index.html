سونەتەکانی سەفەر کردن
١- وا به‌باش داده‌نرێت (سونه‌ته‌) كه‌سه‌فه‌ر كردن له‌ڕۆژی پێنجشه‌مه‌دا بێت و به‌یانی زوو بێت. ٢- وا سونه‌ته‌ موسڵمان پێش ئه‌وه‌ی ده‌ربچێت دوو ڕكات نوێژ بكات و له‌ ڕكاتی یه‌كه‌م سوڕه‌تی (الكافرون) (بِِسْمِ اللَّهِ الرَّحْمَنِ الرَّحِيمِ قُلْ يَاأَيُّهَا الْكَافِرُونَ(١)لاَ أَعْبُدُ مَا تَعْبُدُونَ(٢)وَلاَ أَنْتُمْ عَابِدُونَ مَا أَعْبُدُ(٣)وَلاَ أَنَا عَابِدٌ مَا عَبَدتُّمْ(٤)وَلاَ أَنْتُمْ عَابِدُونَ مَا أَعْبُدُ(٥)لَكُمْ دِينُكُمْ وَلِيَ دِينِ(٦) بخوێندرێت و له‌ ڕكاتی دووه‌م سوره‌تی (الاخلاص) (بِسْمِ اللَّهِ الرَّحْمَنِ الرَّحِيمِ قُلْ هُوَ اللَّهُ أَحَدٌ(١)اللَّهُ الصَّمَدُ(٢)لَمْ يَلِدْ وَلَمْ يُولَدْ(٣)وَلَمْ يَكُنْ لَهُ كُفُوًا أَحَدٌ(٤) بخوێندرێت. حوكمه‌كانی كۆكردنه‌وه‌و كورتكردنه‌وه‌و پێشخستن و دواخستنی نوێژ (جمع و تقصیر) له‌سه‌فه‌ردا كۆكردنه‌وه‌ی نوێژ: له‌سه‌فه‌ری حه‌ج و عومڕه‌دا كاتێ حاجی ده‌چن بۆ عه‌ڕه‌فه‌ واباشه‌ نوێژی عه‌سر له‌گه‌ڵ نوێژی نیوه‌ڕۆدا به‌ (جمع التقدیم) كۆ بكرێته‌وه‌، نوێژی مه‌غریب ببردرێته‌ گه‌ر نوێژی عیشا (جمع التاخیر)و له‌موزده‌لیفه‌ بكرێت، هه‌روه‌كو پێغه‌مبه‌ر (صلی الله علیه وسلم) وای كردووه‌. كورتكردنه‌وه‌: تایبه‌ته‌ به‌نوێژه‌ چوار ڕكاتیه‌كان. وه‌كو: نوێژی نیوه‌ڕۆ و عه‌سرو عیشا. مه‌رجه‌كانی كۆكردنه‌وه‌و كورتكردنه‌وه‌ی نوێژ مه‌رجه‌كانی به‌پێشخستنی نوێژ (جمع التقدیم) ١- به‌دوای یه‌كداهاتن (ترتیب): واته‌ ده‌بێ موسوڵمان له‌پێشدا ئه‌و نوێژه‌ بكات كه‌كاتی خۆیه‌تی و پاشان نوێژه‌كه‌ی تر ئه‌نجام بدات. ٢- نیه‌تی ئه‌وه‌ بهێنێ كه‌ نوێژی دووه‌م له‌گه‌لأ یه‌كه‌م ده‌كات له‌دڵی خویدا پێش ئه‌وه‌ی كه‌ نوێژی یه‌كه‌م ته‌واو بێت. ٣- كه‌له‌نوێژی یه‌كه‌م ته‌واو بوو ده‌بێ یه‌كسه‌ر نوێژی دووه‌م ئه‌نجام بدات به‌بێ دواكه‌وتن. نوێژ دواخستن (جمع التأخیر): ته‌نها مه‌رجی ئه‌وه‌یه‌ كه‌ له‌كاتی نوێژه‌كه‌دا نیه‌تی تاخیر كردن بهێنیت.

‎ ٢- نیەتی ئەوە بێنێ کە نوێژی دوومیان لەگەڵا بە کەم دەکات لەولی خۆیدا پێش ئەوەی کە نوێژی یەکەم نەداو بێت.
‎ ٣- کەلەنوێژی یەکەم تەواو بو دەبێ یەکسەر نوێژی دوومیان ئەنجام بدات بەبی دواکەوتن.

حوكمه‌كانی حه‌ج و عومڕه‌و گه‌وره‌ییان
له‌م باره‌یه‌وه‌ پێغه‌مبه‌ری گه‌روه‌مان (صلی الله علیه وسلم) فه‌رمووده‌ی زۆره‌. یه‌ك له‌وانه‌: جارێكیان دایكی ئیمانداران عائیشه‌ (رض) لێی پرسی ئه‌ی پێغه‌مبه‌ری (صلی الله علیه وسلم) خوا غه‌زا كردن له‌هه‌موو كاره‌كان خێرتره‌ ئه‌ی ئێمه‌ش غه‌زا نه‌كه‌ین؟ پێغه‌مبه‌ر (صلی الله علیه وسلم) له‌وه‌ڵامدا فه‌رمووی: نه‌خێر به‌ڵكو چاكترینی غه‌زا كردن حه‌جێكه‌ كه‌ به‌ڕێك و پێكی ئه‌نجامدرابێت. حه‌ج و عومڕه‌ هه‌ردووكیان فه‌رزن و ڕوكنێكن له‌ ڕوكنه‌كانی ئیسلام به‌ڵگه‌ش له‌م باره‌یه‌وه‌ له‌قورئانی پیرۆز و فه‌رمووده‌كانی پێغه‌مبه‌ر (صلی الله علیه وسلم) و هه‌روه‌ها ئیجماعی زانایان زۆرن. له‌قورئانی پیرۆ خوای گه‌وره‌ ده‌فه‌رموێت: ((وَلِلَّهِ عَلَى النَّاسِ حِجُّ الْبَيْتِ مَنْ اسْتَطَاعَ إِلَيْهِ سَبِيلاً وَمَنْ كَفَرَ فَإِنَّ اللَّهَ غَنِيٌّ عَنْ الْعَالَمِينَ)) (آل عمران:٩٧) دروسته‌ له‌جیات ئافره‌ت و پیاوی پیر فه‌ریزه‌ی حه‌ج ئه‌نجام بدرێت، ئه‌گه‌ر نه‌یتوانی خۆیان ئه‌نجامی بده‌ن، دروست نیه‌ ئافره‌ت به‌ته‌نها بێته‌ ده‌ره‌وه‌ بۆ حه‌ج یان عومڕه‌ بێ ئه‌وه‌ی كه‌سێكی مه‌حڕه‌می له‌گه‌ڵدابێت. میقات: كات و شوێنی دیاریكراو بۆ نیه‌تهێنان ئیحرام به‌ستن به‌مه‌به‌ستی حه‌ج یان عومڕه‌. میقاتی كاتی: مانگه‌كانی ئیحرام به‌ستن بۆ حه‌ج بریتین له‌ شه‌وال و ژوالقعده‌ ده‌ ڕۆژی یه‌كه‌م له‌مانگی ژولحجه‌، به‌ڵام عومڕه‌ له‌هه‌ر كاتێكی ساڵدا بێت ده‌توانرێ نیه‌تی لێ بهێندرێ و ئه‌نجام بدرێت، ته‌نها ئه‌و ڕۆژانه‌ نه‌بێت كه‌ تێكه‌ڵاوی له‌گه‌لأ كاره‌كانی حه‌جدا ڕوو ده‌دات. میقاتی شوێنی (جێ میقات): جێمیقات بۆ خه‌ڵكی مه‌ككه‌، كه‌ ده‌یانه‌وێ حه‌ج بكه‌ن ماڵی خۆیانه‌و له‌ماڵه‌وه‌ ئیحڕام ده‌به‌ستن بۆ ئه‌و كه‌سانه‌ی كه‌ خه‌ڵكی مه‌ككه‌ نین. جێ میقاتی حه‌ج و عومڕه‌ چه‌ند شوێنێكی دیاریكراوه‌ به‌پێی ئاڕاسته‌ی سه‌فه‌ركردنی حاجیه‌كان. بۆ حاجیانی و عومڕه‌كارانی كوردستان، كه‌ له‌ڕێگای مه‌دینه‌ی پیرۆزه‌وه‌ دێنه‌ مه‌ككه‌، ده‌بێت له‌ (ابیار علی) ئیحرام ببه‌ستن و نیه‌ت بێنن. حاجی له‌میقات چی ده‌كات؟ له‌میقات حاجی به‌رگی ئیحرام ده‌كاته‌ به‌رو سوننه‌ته‌ خۆی بشوات و بۆنی خۆی خۆش بكات، هه‌ر وه‌كو پێغه‌مبه‌ر (صلی الله علیه وسلم) كردویه‌تی، واجبه‌ له‌و كاته‌ی كه‌حاجی له‌ژێر ئیحرامدایه‌ سمێڵی نه‌قرتێنێت و نینوكی نه‌قرتێنێت و موو له‌هیچ شوێنێكی له‌شی نه‌كاته‌وه‌، ئه‌گه‌ر توشی یه‌كێك له‌م كارانه‌ش بوو ئه‌وا ده‌بێت فیدیه‌ بدات. ئه‌و شتانه‌ی كه‌ نابێت حاجی به‌هیچ جۆرێك بیانكات (قه‌ده‌غه‌ كراوه‌كانی كاتی حه‌ج كردن) ١- له‌به‌ر كردنی جل و به‌رگ: بۆ پیاوان ده‌بێت سه‌ریان به‌ده‌ره‌وه‌بێت، ئافره‌تیش ده‌بێت ڕووی به‌ده‌ره‌وه‌بێت، له‌و كاته‌ی كه‌ پیاو له‌ژێر ئیحرامدایه‌ نابێ به‌هیچ شێوه‌یه‌ك سه‌ری خۆی دابپۆشێت و ئافره‌تیش نابێ ڕووی خۆی دابپۆشێت. ٢- خۆبۆنخۆشكردن: له‌كاتی ئیحرامدا نابێ حاجی خۆی و جل و به‌رگی یان شوێنی خه‌وتنی بۆنخۆش بكات. ٣- نابێ مرۆڤی ئیحرام به‌ستوو قژ و ڕیشی خۆی چه‌ور بكات (ڕۆنی لێ بدات). ٤- نینۆك قرتاندن یان لابردنی موو له‌هه‌ر به‌شێكی له‌ش و به‌هه‌ر جۆرێك بێت قه‌ده‌غه‌یه‌. ٥- ماره‌بڕین حه‌رامه‌: ئه‌نجامدانی پڕۆسه‌ی هاوسه‌رگیری له‌و كاته‌ی حاجی له‌ژێر ئیحڕامدایه‌ حه‌ڕامه‌و ئه‌گه‌ر ئه‌نجامیشدرا ئه‌وا قه‌بوول نیه‌و پوچه‌ڵه‌. ٦- جیماع و پێشه‌كیه‌كانی به‌هه‌موو شێوه‌یه‌ك حه‌ڕامه‌. ڕاوكردن و كوشتنی نێچیر له‌كاتی ئیحرام به‌ستندا قه‌ده‌غه‌یه‌، چونكه‌ خوای گه‌وره‌ له‌سوڕه‌تی (المائدة‌:٩٥) ده‌فه‌رمووێ: (يَاأَيُّهَا الَّذِينَ آمَنُوا لاَ تَقْتُلُوا الصَّيْدَ وَأَنْتُمْ حُرُمٌ). بۆیه‌ ئه‌گه‌ر هاتوو حاجی ئیحرام به‌ستوو نێچیرێكی كوشت ئه‌وا ده‌بێت فیدیه‌ بدات. ٧- نابێت حاجی ئیحڕام به‌ستو هیچ ڕوه‌كێكی خۆرسك بقرتێنێت، ته‌نانه‌ت ئه‌گه‌ر دڕكیش بێت. هه‌موو ئه‌و قه‌ده‌غه‌كراوانه‌ بۆ ئافره‌ت و پیاو وه‌كو یه‌ك وان.

التشریق) كه‌ ده‌كاته‌ سێیه‌می جه‌ژن كاره‌كه‌ت ته‌واو كرد پێویست ناكات شه‌وی سێیه‌م له‌وێ بمێنیته‌وه‌، هه‌ر وه‌كو خوای گه‌وره‌ ده‌فه‌رمووێت (ََاذْكُرُوا اللَّهَ فِي أَيَّامٍ مَعْدُودَاتٍ فَمَنْ تَعَجَّلَ فِي يَوْمَيْنِ فَلاَ إِثْمَ عَلَيْهِ وَمَنْ تَأَخَّرَ فَلاَ إِثْمَ عَلَيْهِ لِمَنْ اتَّقَى وَاتَّقُوا اللَّهَ وَاعْلَمُوا أَنَّكُمْ إِلَيْهِ تُحْشَرُونَ) (البقرة:٢٠٣) دروسته‌ هه‌ر له‌و ڕۆژه‌شدا ده‌ست بكه‌یت به‌ئه‌نجامدانی عومڕه‌ بۆ ئه‌وه‌ش له‌شی خۆت بشۆ له‌شوێنی مانه‌وه‌ت، یان له‌حه‌مامه‌كانی مزگه‌وتی حه‌زره‌تی عائیشه‌ (رض) پاشان ئیحڕام بكه‌به‌رو دوو ڕكات نوێژ بكه‌ به‌نیه‌تی ئیحڕام و بڵێ: نیه‌تم وایه‌ چوومه‌ ناو عومڕه‌ خودایه‌ لێمی قبوڵ بكه‌، (لبیك اللهم لبیك.. .. . ) پاشان دێیته‌ مه‌ككه‌ی پیرۆز ده‌ست ده‌كه‌ی به‌ته‌واف حه‌وت جار، ئینجا دوو ڕكات سونه‌تی ته‌واف ده‌كه‌یت و پاشان ده‌چیت بۆ سه‌فاو مه‌ڕوا حه‌وت جار له‌نێوانیان دێت و ده‌چیت پاشان قژتاشین یان كورت كردنه‌وه‌ ئه‌نجام ده‌ده‌یت (حلق)، به‌م كارانه‌ عومڕه‌كه‌ت ته‌واو ده‌بێت و هیچ فیدیه‌و كه‌فاڕه‌تێكت له‌سه‌ر نیه‌ ته‌نها له‌حاله‌تێك دانه‌بێت كه‌كارێكی حه‌ڕامت ئه‌نجامدابێت یان واجبێكت جێهێشتبێ ئه‌و كاته‌ فیدیه‌ت له‌سه‌ر واجبه‌. جۆری دووه‌م: ئیحرام به‌ستن به‌عومڕه‌ (التمتع) برای حاجی كاتێ گه‌یشتیته‌ میقات و ویستت ئیحرام ببه‌ستی ته‌نها به‌عومڕه‌ ئه‌وا خۆت خاوێن ده‌كه‌یه‌وه‌و له‌شی خۆت ده‌شۆی و خۆت بۆنخۆش ده‌كه‌یت به‌رگی ئیحرام ده‌كه‌یته‌به‌رو ده‌ڵێی: نیه‌تم وایه‌ چوومه‌ ناو عومڕه‌ خودایه‌ لێم قبول كه‌ی (لبیك اللهم لبیك...). پاشان ده‌چیته‌ مه‌ككه‌ی پیرۆز وه‌ له‌مزگه‌وتی حه‌ڕه‌م حه‌وت كه‌ڕه‌ت سوڕانی ته‌واف ده‌كه‌یت و ئه‌م ته‌وافه‌ ڕوكنی عومڕه‌كه‌ پێك دێنێ و هه‌روه‌ها جێی ته‌وافی (قدوم)یش ده‌گرێته‌وه‌، پاشان دوو ڕكات نوێژی ته‌واف ده‌كه‌یت و ده‌چیت بۆ سه‌فاو مه‌ڕواو حه‌وت جار دێیت و ده‌چیت، ئینجا قژ ده‌تاشی یان كورتی ده‌كه‌یته‌وه‌و به‌مه‌ش عومره‌كه‌ت ته‌واوده‌بێت و به‌رگی خۆت ده‌كه‌یته‌وه‌ به‌ر له‌م حاله‌ته‌دا واته‌ له‌ته‌موتوعدا فیدیه‌ واجب ده‌بێت، كه‌پێشی ده‌وترێت (خوێن) له‌ڕۆژی الترویه‌ (ڕۆژی پێش عه‌ره‌فه‌) نیته‌ حه‌ج دێنیت و ئیحرام ده‌به‌ستی و كاره‌كانی حه‌ج یه‌ك به‌دوای یه‌ك ئه‌نجام ده‌ده‌یت، هه‌روه‌ك پێشتر باسكراوه‌. جۆری سێیه‌م له‌ئیحرام به‌ستن بریتیه‌ له‌ دووانه‌كی (القران) ئه‌ی حاجی به‌ڕێز كاتێ گه‌یشتیه‌ یه‌كێ له‌میقاته‌كان و ویستت ئییحرام ببه‌ستیت به‌قیڕان واته‌ حه‌ج و عومڕه‌ پێكه‌وه‌ ئه‌وا خۆت خاوێن بكه‌وه‌و له‌شی خۆت ده‌شۆی و خۆت بۆنخۆش ده‌كه‌یت به‌رگی ئیحرام دوڕكات نوێژی ئیحرام بكه‌و، كه‌ سه‌لامت دایه‌وه‌ بڵێ: نیه‌تم وایه‌ چوومه‌ ناو حه‌ج و عومڕه‌ خودایه‌ لێم قبول كه‌ی (لبیك اللهم لبیك....) دوای ئه‌وه‌ كاره‌كان ئه‌نجام ده‌ده‌یت هه‌روه‌كو له‌حاله‌تی ئیحرام به‌ستن به‌حه‌ج (افراد)وه‌ له‌م حاله‌ته‌دا یه‌ك ته‌واف (حه‌وت خولی ته‌واو)و یه‌ك هاتوچۆی سه‌فاو مه‌ڕوا (حه‌وت جاره‌كی) به‌سه‌ بۆ حه‌ج و عومڕه‌ هه‌ردووكیان، پاشان ده‌بێت ئاژه‌ڵێك بكوژیته‌وه‌، به‌ڵام ئه‌گه‌ر نه‌یتوانی ده‌بێت ده‌ ڕۆژ به‌ڕۆژوو ببێت، سێ ڕۆژ له‌كاتی حه‌ج و حه‌وتیش له‌كاتی گه‌ڕانه‌وه‌.

ئه‌رك و ڕوكنه‌كانی حه‌ج
باسی یه‌كه‌م چۆنیه‌تی ته‌واف كردن كاتێ حاجی ئیحرام له‌به‌ر گه‌یشته‌ مه‌كه‌ی پیرۆز سونه‌ت وایه‌ خۆی بشوات و كاتێ گه‌یشته‌ حه‌ڕه‌می پیرۆز سونه‌ته‌ پێی ڕاسته‌ی پێش بخاته‌ ژووره‌وه‌و بڵێت (أعوذ بالله العظيم و بوجهه الكريم وسلطان القجيم من الشيطان الرجيم اللهم افتح لى ابواب رحمتك). كاتێ حاجی یان عومڕه‌كار ده‌گاته‌ حه‌ڕمی مه‌ككه‌ دبێت ڕوو بكاته‌ حه‌جه‌رولئه‌سوه‌د، كه‌ده‌كه‌وێته‌ گۆشه‌ه‌ی دوای ده‌رگای به‌یت له‌لای ڕۆژهه‌ڵاته‌وه‌، پێی ده‌وترێت (الركن الاسود) واخێره‌ كه‌وا له‌حه‌جه‌ڕولئه‌سوه‌د نزیك بێته‌وه‌و ڕووی تێ بكات به‌مه‌رجێك ئازاری خه‌لك نه‌دات ده‌ستی لێبداو پاشان به‌بێ ده‌نگی ماچی بكات، ئینجا ده‌ست به‌ته‌واف بكات و له‌كاتی ته‌وافدا (لبیك) ناڵێت، بۆ پیاوان له‌گه‌لأ ده‌ست پێ كردنی ته‌وافدا واخێره‌ شانی ڕاسته‌ی رووت بكاته‌وه‌، بۆ ئه‌نجامدانی ته‌واف ده‌بێ هه‌موو له‌شی به‌ته‌واوی روو بكاته‌ حه‌جه‌رولئه‌سوه‌دو به‌رامبه‌ری بێتو خۆی ده‌سوڕێنێت تا شانی ڕاسته‌ی به‌رامبه‌ری حه‌جه‌رولئه‌سوه‌د ده‌بێت ئینجا نیه‌ت دێنێت و پاشان به‌رامبه‌ر حه‌جه‌رولئسوه‌د خۆی ده‌جولێنێته‌وه‌ به‌ره‌و لای ڕاستی خۆی هه‌تا له‌ڕاستی حه‌جه‌رولئه‌سوه‌د تێپه‌ڕ ده‌بێت، ئینجا لای چه‌پی خۆی ده‌كات به‌یت و لای راستیشی ڕوو له‌ده‌ره‌وه‌ ده‌بێت به‌م جۆره‌ ده‌ڕوات هه‌تا حه‌وت جار له‌ده‌وره‌ی به‌یت ده‌سوڕێته‌وه‌. حاجی كاتێ ده‌ستی كرد به‌ته‌واف ده‌ڵێ (بسم الله الله اكبر) و بۆ پیاوان سونه‌ته‌ له‌سێ كه‌ڕه‌تی یه‌كه‌م له‌ته‌واف به‌خێرایی بڕوات و شانی ڕاسته‌ی به‌ده‌ره‌وه‌ بێت و ئه‌م دوو سونه‌ته‌ ته‌نها له‌یه‌كه‌م ته‌وافدا (طوف القدوم) په‌یڕه‌و ده‌كرێت. ته‌واف كۆمه‌ڵێك مه‌رج و واجباتی هه‌یه‌ به‌ببێ ئه‌وانه‌ ته‌وافه‌كه‌ دروست نابێت، كه‌بریتین له‌: داپۆشین و شاردنه‌وه‌ی عه‌وڕه‌ت و پاكژی له‌پیسی، واته‌ نابێ له‌شی پیس بێت هه‌روه‌ها پاكژی له‌بێ ده‌ست نوێژی، پاكژی جل و به‌رگ وئه‌و شوێنه‌ی كه‌پێی له‌سه‌ر داده‌نێت، ئه‌گه‌ر جاجی له‌كاتی ته‌واف كردندا شوێنێك له‌عه‌وره‌تی ده‌ركه‌وت یان پیسی كه‌وته‌ سه‌ر له‌شی یان جل و به‌رگی یان به‌ئه‌نقه‌ست به‌سه‌ر پیساییدا تێپه‌ڕی و خۆی خاوێن نه‌كرده‌وه‌ یان له‌بیری چوو، ئه‌وا ته‌وافه‌كه‌ی به‌رناكه‌وێت، ئه‌و ئافره‌تانه‌ی پێیان به‌ده‌ره‌وه‌بێت یان به‌شێك له‌سه‌ریان ڕووت بێت ته‌وافه‌كه‌یان ته‌واو نیه‌، ئه‌گه‌ر ته‌نانه‌ت یه‌ك مووی سه‌ریان یان یه‌ك نینۆكی پێیان به‌ده‌ره‌وه‌ بێت ته‌وافه‌كه‌یان به‌تاڵه‌. ته‌واف كردن به‌ده‌وری كه‌عبه‌ی پیرۆز وه‌كو نوێژ وایه‌ به‌ڵام حاجی ده‌توانێت لوه‌كاتی ته‌واف كردندا قسه‌ بكات به‌مه‌رجێك ته‌نها به‌چاكه‌ بدوێت هه‌روه‌كو له‌فه‌رمووده‌ی پێغه‌مبه‌ردا صلی الله علیه وسلم هاتووه‌ (الطواف بالبيت صلاة الاان الله قد اباح لكم فيه الكلام فمن تكلم فلا يتكلم الابخير). باسی دووه‌م سه‌عی (سه‌فاو مه‌ڕوا) كاتێك حاجی ته‌وافی ته‌واو كردو دوو ڕكات سونه‌ته‌كه‌ی كرد ئه‌وا سونه‌ت وایه‌ بچێته‌ به‌رامبه‌ر حه‌جه‌رولئه‌سوه‌د و ده‌ستی بۆ به‌رزكاته‌وه‌ (به‌شێوه‌ی سه‌لام كردن) پاشان ده‌ڕوات به‌ره‌و سه‌فا بۆ سه‌عی كردن ده‌چێته‌ سه‌ر به‌رزای سه‌فا به‌قه‌د به‌ژنێك تا كه‌عبه‌ی پیرۆز ده‌بینێت ئینجا ده‌ڵێت (الله اكبر الله اكبر الله اكبر ولله الحمد الله اكبر على ماهدانا والحمدلله على ما اولانا لااله الا الله وحده لاشريك له له المللك وله الحمد يحى ويميت بيده الخير وهو على كل شيء قدير الااله الا الله وحده لاشريك له انجز وعده ونصر عبده وهزم الاحزاب وحده الاله الاالله ولا نعبد الا اياه مخلصين له الدين ولوكره الكافرون) پاشان به‌ئاره‌زووی خۆی دوعا ده‌كات ئینجا له‌سه‌فا دێته‌ خواره‌وه‌و ده‌ست به‌ڕۆیشتن ده‌كات به‌ره‌و مه‌ڕوا، له‌نێوان دوو گلۆپه‌ سه‌وزه‌كه‌ به‌خێرایی ده‌ڕوات. له‌عیمڕان ده‌گێڕنه‌وه‌ده‌ڵێت پێغه‌مبه‌ر صلی الله علیه وسلم هات بۆ به‌یت حه‌وت جار به‌ده‌وری دا سوڕایه‌وه‌ وه‌ دوو ڕكات نوێژی كرد له‌پشت مه‌قامی ابراهیم و حه‌وت جاریش له‌نێوان سه‌فاو مه‌ڕوا هات و چو وه‌پێغه‌مبه‌ریش صلی الله علیه وسلم پێشه‌وامانه‌و ده‌بێ چی كردووه‌و وه‌ك ئه‌و بكه‌ین. زیكرو دوعا له‌كاتی ته‌واف و سه‌عی دا. له‌ عبدالله كوڕی سائیب ده‌گێڕنه‌وه‌ گوتوویه‌تی گوێم لێ بوو پێغه‌مبه‌ری صلی الله علیه وسلم له‌نێوان ڕووكنی یه‌مانی و حه‌جه‌رولئه‌سوه‌دا دیفه‌رموو (رَبَّنَا آتِنَا فِي الدُّنْيَا حَسَنَةً وَفِي الاَخِرَةِ حَسَنَةً وَقِنَا عَذَابَ النَّارِ) پاشان وتیان ئه‌ی پێغه‌مبه‌ری خوا صلی الله علیه وسلم له‌كاتی گه‌یشتن به‌حه‌جه‌رولئه‌سوه‌دا چی بڵێن فه‌رمووی بڵێن (بسم الله الله اكبر ايمانا باالله وتصديقا بما جاء به محمد). واجب و مه‌رجه‌كانی سه‌عی سه‌فاو مه‌ڕوا. واجبی یه‌كه‌م: ده‌بێت حاجی هه‌موو مه‌ودای نێوان سه‌فاو مه‌ڕوا ببڕێت ئه‌گه‌ر چه‌ند هه‌نگاوێك جێ بهێڵێت ئه‌وا ئه‌وا سه‌عیه‌كه‌ی ناته‌واوه‌. واجی دووه‌م: ڕیزبه‌ندی (الترتیب): واجبه‌ حاجی له‌ سه‌فاوه‌ ده‌ست بكات به‌ڕۆیشتن، ئه‌گه‌ر له‌مه‌ڕوه‌ ده‌ستی پێكرد هه‌تا نه‌گاته‌ سه‌فای بۆی حساب نیه‌. واجبی سێیه‌م: ته‌واو كردنی حه‌وت هاتوچۆی ته‌واو له‌نێوان سه‌فاو مه‌ڕوادا له‌سه‌فاوه‌ تامه‌ڕوا به‌جارێك داده‌نرێت و گه‌ڕانه‌وه‌ش له‌مه‌ڕوا بۆ سه‌فا به‌جارێك. واجبی چواره‌م: ده‌بێتهاتوچۆی نێوان سه‌فاو مه‌ڕوا له‌ دوای ته‌وافبَكی ڕاست ئه‌نجام بدرێت. سونه‌ته‌كانی سه‌عی (سه‌فاو مه‌ڕوا). یه‌كه‌م: زیكرو دوعا كردن له‌ سه‌فاو مه‌ڕوا. دووه‌م: پاك و خاوێنی و داپۆشینی عه‌وره‌ت. سێیه‌م: خێرا ڕۆیشتن له‌نێوان هه‌ردوو گلۆپه‌ سه‌وزه‌كه‌. چواره‌م: سونه‌ته‌ له‌كاتێكی وادا سه‌فاو مه‌ڕوا بكرێت كه‌ چۆل بێت بۆ ئه‌وه‌ی خه‌ڵكی توشی ئازار نه‌بێت. پێنجه‌م: سونه‌ته‌ هه‌ر حه‌وت جاره‌كه‌ به‌دوای یه‌ك دابێن و نێوانیان نه‌بێت. باسی سێیه‌م:وه‌ستان له‌عه‌ره‌فه‌. كاتێ حاجی سه‌فاو مه‌ڕوا ته‌واو ده‌كات ئه‌گه‌ر بۆ عه‌مڕه‌ بێت ئه‌وا پێویسته‌ چه‌ند موویه‌ك له‌سه‌ری ببڕێت، یان كورت بكاته‌وه‌، پاشان ئه‌گه‌ر ئیحرامی به‌عومڕه‌ بستبێت ئه‌وا له‌مه‌كه‌ ده‌مێنێته‌وه‌ به‌حه‌لالی واته‌ هه‌موو ئه‌و شتانه‌ی كه‌له‌كاتی ئیحرام به‌ستندا لێی حه‌رام بوو بۆی حه‌لال ده‌بێت، هه‌روه‌ها ده‌توانێت عومره‌ی تر بهێنێته‌وه‌ ئه‌گه‌ر ویستی، له‌ڕۆژی هه‌شته‌می ژلحجه‌ (يوم التروية‌)ڕۆژی ته‌رویه‌ حاجی له‌ مه‌ككه‌ ئیحرام ده‌به‌ستێت خۆرهه‌ڵاتنی ڕۆژی جه‌ژن ده‌ست پێ ده‌كات به‌ڵاَم لای شافعیه‌كان دوای نیوه‌ی شه‌وی جه‌ژن ده‌ست پێ ده‌كات كۆتایی دێت له‌دواهه‌مین ڕۆژی (ایام التشریق) ده‌بێت ڕه‌می حه‌وت به‌رد بێت و له‌گه‌لأ هاویشتنی هه‌ر كردبێت له‌واجبه‌كانی حه‌ج. ٤- ئه‌و حاجیانه‌ی تاوانێكیان ئه‌نجامدابێ كه‌پێویستی به‌فیدیه‌ بێت. ئه‌گه‌ر ئه‌م حاله‌تانه‌ نه‌بوو، ئه‌وا كوشتنه‌وه‌ی ئاژه‌ڵ سونه‌ته‌. باسی شه‌شه‌م قژ تاشین یان كورت كردنه‌وه‌: لای شافعیه‌كان ڕوكنێكی سه‌ره‌كیه‌و كاته‌كه‌ی له‌ناوه‌ڕاستی شه‌وی جه‌ژنه‌وه‌ ده‌ست پێ ده‌كات و هه‌تتا ئه‌و كاته‌ی كه‌ حاجی له‌ژێر ئیحرام دێته‌ ده‌ره‌وه‌ به‌رده‌وام ده‌بێت بۆ ئه‌م مه‌به‌سته‌ش باشترین كاتیش بۆ ئه‌نجامدانی ڕۆژی جه‌ژنه‌، به‌ڵام دواخستی بۆ ڕۆژانی دوای جه‌ژن به‌باش دانانرێت ئه‌گه‌ر ته‌وافه‌كه‌ی ته‌وافی وه‌داع بوو ته‌وافی (الافاضة)شی نه‌كردبوو، ئه‌وا ته‌وافه‌كه‌ بۆ به‌ته‌وافی ده‌یه‌وێت مه‌ككه‌ به‌جێبهێڵێت و زۆربه‌ی زانایان وای داده‌نێت واجبه‌، كه‌سێك ته‌وافی وه‌داعی له‌سه‌ر واجب بێت و بڕوات به‌بێ ئه‌وه‌ی ئه‌نجامی بدات گوناهی پێ ئه‌گات و ده‌بێت بۆ بگه‌ڕێته‌وه‌ به‌ڵامم یه‌كه‌م: عومڕه‌ واجبه‌ له‌سه‌ر كه‌سێ كه‌توانای هه‌بێت وه‌كو حه‌ج له‌لای شافعیه‌كان و له‌ژیاندا یه‌كجار واجبه‌ ئه‌گه‌ر زیاتر له‌جارێكیش بێت ئه‌وا خێره‌، به‌تایبه‌تی له‌مانگی ڕه‌ه‌مه‌زان هه‌روه‌كو له‌حه‌دیسی راستدا هاتووه‌ كه‌ پێغه‌مبه‌ر صلی الله علیه وسلم فه‌رموویه‌تی له‌عومره‌یه‌كه‌وه‌ بۆ عومڕه‌یه‌كی تركه‌فاره‌تی گوناهه‌كانه‌ له‌ابن عباس ده‌گێڕنه‌وه‌ (علیه وسلم) كه‌ پێغه‌مبه‌ر صلی الله علیه وسلم فه‌رموی: ئه‌نجامدانی عومڕه‌یه‌ك له‌ڕه‌مه‌زاندا هاوتای حه‌جێكه‌ له‌گه‌لأ من (عمره‌ فی رمچان كحجه‌ معی). دووه‌م: بۆ عومڕه‌ی به‌ته‌نها دوو میقات هه‌یه‌، میقات كاتی و میقاتی شوێنی هه‌روه‌ك هی حه‌ج وایه‌، میقاتی كاتی به‌درێژایی ساڵ هه‌ر كاتێك بته‌وێت كاتی ئیحرام به‌ستنه‌ ته‌نها ئه‌و كه‌سه‌ نه‌بێت كه‌له‌كاتی حه‌جدا ئیحرامی به‌حه‌ج به‌ستوه‌. كاره‌كانی عومڕه‌. له‌ئیحرام به‌ستن بۆ عومڕه‌ هه‌ر وه‌كو حه‌ج، خۆ شوشتن و بۆن خۆش كردن و خۆ پاككردنه‌وه‌ سونه‌تن، كاتی ئیحرام به‌ستن دوو ڕكات سونه‌ت ده‌كات و ئیحرام به‌عومڕه‌ ده‌به‌ستێت و به‌رده‌وام (لبیك.. .. ) ده‌ڵێت تا ده‌گاته‌ مه‌كه‌ی پیرۆز له‌وێ ده‌ست ده‌كات به‌ئه‌نجامدانی ته‌واف و (لبیك.. .. . ) ناڵێت له‌سێ خولی یه‌كه‌م خێرا ده‌ڕوات له‌چواره‌كه‌ی تر به‌ئاسایی ڕۆیشتن ده‌كات پاشان دده‌چێت هاتوچۆی سه‌فاو مه‌ڕوا ئه‌نجام ده‌دات. ئه‌ركه‌كانی عومڕ چوارن: ١- ئیحرام به‌ستن. ٢- ته‌واف كردن. ٣- سعی (هاتو چۆی نێوان سه‌فاو مه‌ڕوا). ٤- قژ بڕین. ٥- فیدیه‌ واجب ده‌بێت له‌كاتی ئه‌نجامدانی كارێك كه‌ به‌هۆی ئیحڕام به‌ستنه‌وه‌ حه‌رام كرابێت ئه‌گه‌ر به‌نه‌زانیش بێ یان به‌هۆی له‌بیرچونه‌وه‌ بێ به‌م شێوه‌یه‌ی خواره‌وه‌: ١- به‌لابردنی سێ موو یان قرتاندنی سێ نینۆك ئاژه‌ڵێك واجب ده‌بێت، به‌ڵام به‌لابردنی یه‌ك موو یان یه‌ك نینۆك فیدیه‌كه‌ی ته‌نها مشتێك (مستيَك) خواردنه‌ بۆ دوانیش دوو مشته‌. ٢- هه‌ر كه‌سێك یه‌كێك له‌عیباده‌ته‌كانی حه‌ج و عومڕه‌ی به‌تاڵ كردبێته‌وه‌ به‌جیماع كردن ئه‌وا ده‌بێت ئاژه‌ڵێكی شیاو بۆ قوربانی سه‌ر ببڕێت و به‌رده‌وامیش ده‌بێت له‌عیباده‌ته‌كانی.
